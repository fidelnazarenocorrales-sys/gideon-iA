        function addMessage(role, content, isImage = false, imageUrl = null) {
            const chat = document.getElementById('chat');
            const msg = document.createElement('div');
            msg.className = `flex ${role === 'user' ? 'justify-end' : 'justify-start'} message-fade`;

            if (role === 'user') {
                if (isImage && imageUrl) {
                    msg.innerHTML = `<div class="bubble-user p-2 max-w-[85%]"><img src="${imageUrl}" class="rounded-xl border-2 border-white/20"></div>`;
                } else {
                    msg.innerHTML = `<div class="bubble-user px-6 py-4 max-w-[85%] font-bold text-lg">${content}</div>`;
                }
            } else {
                const htmlContent = marked.parse(content);
                msg.innerHTML = `
                    <div class="flex gap-4 max-w-[90%]">
                        <div class="text-3xl mt-1 text-cyan-500 drop-shadow-[0_0_10px_rgba(6,182,212,0.8)]">◆</div>
                        <div class="bubble-gideon px-7 py-5 flex-1 border border-zinc-700/50 shadow-2xl">
                            <div class="prose prose-invert prose-cyan max-w-none text-zinc-100 text-[18px] leading-relaxed">
                                ${htmlContent}
                            </div>
                            <button onclick="speak('${content.replace(/['"`\n]/g, " ")}')" class="mt-5 text-cyan-500 text-[11px] font-black hover:text-white flex items-center gap-3 uppercase tracking-[0.3em] transition-all opacity-70 hover:opacity-100">
                                <i class="fas fa-volume-up text-base"></i> Transmisión de Audio
                            </button>
                        </div>
                    </div>`;
            }
            chat.appendChild(msg);
            chat.scrollTo({ top: chat.scrollHeight, behavior: 'smooth' });
        }

        async function sendMessage() {
            const input = document.getElementById('userInput');
            const text = input.value.trim();
            if (!text) return;

            addMessage('user', text);
            input.value = '';

            const respuesta = await getGideonResponse(text);
            addMessage('assistant', respuesta);
            speak(respuesta);
        }

        function speak(text) {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();
                const u = new SpeechSynthesisUtterance(text);
                u.lang = 'es-AR';
                u.rate = 1.1;
                speechSynthesis.speak(u);
            }
        }

        function handleImage(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(ev) {
                addMessage('user', '', true, ev.target.result);
                setTimeout(() => {
                    addMessage('assistant', 'Escaneo visual completado. Datos integrados en mi base de datos de análisis.');
                    speak('Escaneo visual completado.');
                }, 800);
            };
            reader.readAsDataURL(file);
        }

        function startVoiceInput() {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            if (!SpeechRecognition) return alert("Navegador no compatible.");
            const recognition = new SpeechRecognition();
            recognition.lang = 'es-AR';
            recognition.onresult = (event) => {
                document.getElementById('userInput').value = event.results[0][0].transcript;
                sendMessage();
            };
            recognition.start();
        }

        window.onload = () => {
            addMessage('assistant', 'Sistemas de Inteligencia Artificial en línea. Saludos, Fidel. He sincronizado mi conciencia con el presente. ¿En qué podemos trabajar hoy?');
        };
